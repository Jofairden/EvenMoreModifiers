56  internal ModifierPool RollNewPool(ModifierContext ctx, ItemRollProperties itemRollProperties = null)

74  noForce = !ModifierPool?._CanRoll(ctx) ?? true;
75  }
76

78  if (noForce)

92  noForce = !rollPredefinedPool;
93 
94  if (rollPredefinedPool)

96  // GetWeightedPool already checks _CanRoll

98  noForce = ModifierPool == null || !ModifierPool._CanRoll(ctx);
99  }
100

112  }

138  internal bool RollNewModifiers(ModifierContext ctx, ItemRollProperties itemRollProperties)
139  {
140  // Firstly, prepare a WeightedRandom list with modifiers
141  // that are rollable in this context
142  WeightedRandom<Modifier> wr = new WeightedRandom<Modifier>();
143  List<Modifier> list = new List<Modifier>();
144 
145  foreach (var e in ModifierPool.RollableModifiers(ctx))
146  {
147  wr.Add(e, e.Properties.RollChance);
148  }
149 
150  // Up to n times, try rolling a mod
151  // @ todo since we can increase lines rolled, make it so that MaxRollableLines influences the number of rows drawn in the UI
152  for (int i = 0; i < itemRollProperties.MaxRollableLines; ++i)
153  {
154  // If there are no mods left, or we fail the roll, break.
155  if (wr.elements.Count <= 0
156  || !_forceNextRoll
157  && i > 0
158  && list.Count >= itemRollProperties.MinModifierRolls
159  && Main.rand.NextFloat() > itemRollProperties.RollNextChance)
160  {
161  break;
162  }
163 
164  _forceNextRoll = false;
165 
166  // Get a next weighted random mod
167  // Clone the mod (new instance) and roll it's properties, then roll it
168  Modifier e = wr.Get();
169  Modifier eClone = (Modifier)e.Clone();
170  float luck = itemRollProperties.ExtraLuck;
171  if (ctx.Player != null)
172  {
173  luck += ModifierPlayer.Player(ctx.Player).GetEffect<LuckEffect>().Luck;
174  }
175  eClone.Properties =
176  eClone.GetModifierProperties(ctx.Item)
177  .RollMagnitudeAndPower(
178  magnitudePower: itemRollProperties.MagnitudePower,
179  lukStat: luck);
180  eClone.Roll(ctx, list);
181 
182  // If the mod deemed to be unable to be added,
183  // Force that the next roll is successful
184  // (no RNG on top of RNG)
185  if (!eClone.PostRoll(ctx, list))
186  {
187  _forceNextRoll = true;
188  continue;
189  }
190 
191  // The mod can be added
192  list.Add(eClone);
193 
194  // If it is a unique modifier, remove it from the list to be rolled
195  if (eClone.Properties.UniqueModifier)
196  {
197  wr.elements.RemoveAll(x => x.Item1.Type == eClone.Type);
198  wr.needsRefresh = true;
199  }
200  }
201 
202  ModifierPool.ActiveModifiers = list.ToArray();
203  return list.Any();
204  }

94  var detachEffects = GetModifierEffectsForDelegations(_detachList, modPlayer, (e) => e.IsBeingDelegated && !_modifierEffects.Contains(e.GetType()));
95  var attachEffects = GetModifierEffectsForDelegations(_attachList, modPlayer, (e) => !e.IsBeingDelegated);
96  // Automatic delegation lists
97  var orderedDetachList = OrderDelegationList(_detachList, modPlayer)
98  .Where(x => x.effect.IsBeingDelegated && !_modifierEffects.Contains(x.effect.GetType()))
99  .GroupBy(x => x.methodInfo)
100  .Select(x => x.First())
101  .ToList();
102 
103  var orderedAttachList = OrderDelegationList(_attachList, modPlayer)
104  .Where(x => !x.effect.IsBeingDelegated)
105  .GroupBy(x => x.methodInfo)
106  .Select(x => x.First())
107  .ToList();
108

111  {
112  modPlayer.OnResetEffects -= effect.ResetEffects;
113  effect._DetachDelegations(modPlayer);
114  effect.IsBeingDelegated = false;
115  }
116

119  {

121  effect.AttachDelegations(modPlayer);
122  effect.IsBeingDelegated = true;
123  }
124

145  private IEnumerable<(MethodInfo methodInfo, ModifierEffect effect)> OrderDelegationList(IEnumerable<(Item item, Modifier modifier)> list, ModifierPlayer modPlayer)
146  {
147  var delegationEntries = new List<(MethodInfo, ModifierEffect)>();
148  var effects = GetModifierEffectsForDelegations(list, modPlayer, e => true);
149 
150  foreach (var modEffect in effects)
151  {
152  (MethodInfo, ModifierEffect) MakeEntry((MethodInfo, DelegationPrioritizationAttribute) tuple)
153  => (tuple.Item1, modEffect);
154 
155  var delegatedMethods = modEffect
156  .GetType()
157  .GetMethods(BindingFlags.NonPublic | BindingFlags.Public | BindingFlags.Instance)
158  .Where(x => x.GetCustomAttributes(typeof(AutoDelegation), false).Length > 0)
159  .Select(x => (methodInfo: x, attribute: (DelegationPrioritizationAttribute) x.GetCustomAttribute(typeof(DelegationPrioritizationAttribute))))
160  .ToList();
161 
162  delegationEntries.AddRange(
163  delegatedMethods
164  .Where(x => x.attribute?.DelegationPrioritization == DelegationPrioritization.Early)
165  .OrderByDescending(x => x.attribute.DelegationLevel)
166  .Select(MakeEntry));
167 
168  delegationEntries.AddRange(
169  delegatedMethods
170  .Where(x => x.attribute == null)
171  .Select(MakeEntry));
172 
173  delegationEntries.AddRange(
174  delegatedMethods
175  .Where(x => x.attribute?.DelegationPrioritization == DelegationPrioritization.Late)
176  .OrderByDescending(x => x.attribute.DelegationLevel)

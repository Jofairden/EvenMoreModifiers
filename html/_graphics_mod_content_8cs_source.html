248  //to clarify, this code simply 'fixes' out bad saving of the file in photoshop or whatever
249  //yeah, it works for every pixel individually
250  //some art programs save the file with every pixel having an RGBA where RGB is bigger than 0 but A may be 0 , where A would usually be transparency there, in XNA's texture loading the alpha channel is independent , and with terraria's draw mode it actually can do awesome things
251  //in terraria, A channel signals additive<---> opaque
252  //a pixel with 255 alpha draws fully opaque, a pixel with 0 alpha draws fully additive
253  //so if you do say, 255,0,0,0 color, it would draw a fully additive red
254  //which can be awesome...except when your file saves the alpha channel of supposedly transparent pixels wrong
255  public static Texture2D MultiplyColorsByAlpha(Texture2D texture)
256  {
257  Color[] data = new Color[texture.Width * texture.Height];
258  texture.GetData(data);
259  for (int i = 0; i < data.Length; i++)
260  {
261  Vector4 we = data[i].ToVector4();
262  data[i] = new Color(we.X * we.W, we.Y * we.W, we.Z * we.W, we.W);
263  }
264 
265  texture.SetData(data);
266  return texture;
267  }
